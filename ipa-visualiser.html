    document.addEventListener('DOMContentLoaded', () => {

        // State

        let clos = []; // [{num, text, verb}]

        let units = []; // [{code, name, level, ulos:[{text,verb,bloomsLevel}], ipaMapping:{cloIndex:'I'|'P'|'A'}, assessments:[string]}]


        // Bloom's verb detection

        const BLOOMS_VERBS = {

            1: ['define','describe','identify','label','list','name','outline','recall','recognise','recognize','state','match','select'],

            2: ['classify','compare','contrast','discuss','distinguish','explain','illustrate','interpret','summarise','summarize','paraphrase','infer'],

            3: ['apply','calculate','carry out','construct','implement','modify','solve','use','demonstrate','execute','operate'],

            4: ['analyse','analyze','categorise','categorize','deconstruct','differentiate','examine','investigate','research','organise','organize','attribute'],

            5: ['appraise','argue','assess','critique','defend','evaluate','justify','recommend','review','judge','monitor'],

            6: ['compose','construct','create','design','develop','formulate','generate','hypothesise','hypothesize','produce','synthesise','synthesize','plan','invent']

        };


        function detectBloomsLevel(text) {

            const lower = text.toLowerCase().trim();

            for (let level = 6; level >= 1; level--) {

                for (let verb of BLOOMS_VERBS[level]) {

                    if (lower.startsWith(verb)) return level;

                }

            }

            // Check first few words

            const firstWords = lower.split(/\s+/).slice(0, 3).join(' ');

            for (let level = 6; level >= 1; level--) {

                for (let verb of BLOOMS_VERBS[level]) {

                    if (firstWords.includes(verb)) return level;

                }

            }

            return 0; // unknown

        }


        function getBloomsLabel(level) {

            const labels = { 1:'Remember', 2:'Understand', 3:'Apply', 4:'Analyse', 5:'Evaluate', 6:'Create' };

            return labels[level] || 'Unknown';

        }


        function getUnitLevel(code) {

            const match = code.match(/(\d)\d{3}/);

            if (match) return parseInt(match[1]) * 1000;

            return 0;

        }


        function getFirstVerb(text) {

            const words = text.trim().split(/[\s,]+/);

            return words[0] || '';

        }


        // Infer IPA based on unit level relative to CLO

        function inferIPA(unitLevel) {

            if (unitLevel <= 1000) return 'I';

            if (unitLevel <= 2000) return 'P';

            return 'A';

        }

                .filter(l => l.trim())

                .map((l, i) => {

                    const cleaned = l.replace(/^\d+[\.\)]\s*/, '').trim();

                    return {

                        num: i + 1,

                    };

            }) : [];

            renderAll();

        });


        // Generate IPA mapping for a unit based on its level

        function generateIPAMapping(unit) {

            unit.ipaMapping = {};

            if (clos.length === 0) return;


            const ipa = inferIPA(unit.level);


            // For each CLO, check if any ULO in this unit relates to it

            // Simple heuristic: if unit has ULOs, try to match keywords

            // If no ULOs, assign all CLOs the inferred IPA level

            if (unit.ulos.length === 0) {

                // Assign all CLOs at this IPA level

                clos.forEach((clo, i) => {

                    unit.ipaMapping[i] = ipa;

                });

            } else {

                // Try to match ULOs to CLOs by keyword similarity

                clos.forEach((clo, i) => {

                    const cloWords = new Set(clo.text.toLowerCase().split(/\W+/).filter(w => w.length > 3));

                    let bestMatch = 0;

                    unit.ulos.forEach(ulo => {

                        const uloWords = ulo.text.toLowerCase().split(/\W+/).filter(w => w.length > 3);

                        const overlap = uloWords.filter(w => cloWords.has(w)).length;

                        if (overlap > bestMatch) bestMatch = overlap;

                    });

                    if (bestMatch >= 2) {

                        unit.ipaMapping[i] = ipa;

                    }

                });


                // If very few matches, fall back to assigning all

                const mappedCount = Object.keys(unit.ipaMapping).length;

                if (mappedCount < clos.length * 0.3) {

                    clos.forEach((clo, i) => {

                        if (!unit.ipaMapping[i]) unit.ipaMapping[i] = ipa;

                    });

                }

            }

        }


        // Paste units modal

        document.getElementById('paste-units-btn').addEventListener('click', () => {

            document.getElementById('paste-modal').style.display = 'flex';

        });

        document.getElementById('paste-units-cancel').addEventListener('click', () => {

            document.getElementById('paste-modal').style.display = 'none';

        });

        document.getElementById('paste-units-confirm').addEventListener('click', () => {

            const data = document.getElementById('paste-units-data').value.trim();

            if (!data) return;


            const lines = data.split('\n').filter(l => l.trim());

            lines.forEach(line => {

                const parts = line.split('\t');

                if (parts.length < 2) return;

                const code = parts[0].trim();

                const name = parts[1].trim();

                const ulosRaw = parts[2] ? parts[2].split(';').filter(u => u.trim()) : [];

                const level = getUnitLevel(code);

                const ulos = ulosRaw.map(u => {

                    const cleaned = u.trim().replace(/^\d+[\.\)]\s*/, '');

                    return { text: cleaned, verb: getFirstVerb(cleaned), bloomsLevel: detectBloomsLevel(cleaned) };

                });

                const unit = { code, name, level, ulos, ipaMapping: {}, assessments: [] };

                generateIPAMapping(unit);

                units.push(unit);

            });


            document.getElementById('paste-modal').style.display = 'none';

            document.getElementById('paste-units-data').value = '';

            renderAll();

        });


        // Cycle IPA value on click

        function cycleIPA(unitIdx, cloIdx) {

            const unit = units[unitIdx];

            const current = unit.ipaMapping[cloIdx] || '';

            const cycle = { '': 'I', 'I': 'P', 'P': 'A', 'A': '' };

            unit.ipaMapping[cloIdx] = cycle[current];

            renderAll();

        }


        // Delete unit

        function deleteUnit(idx) {

            if (confirm('Remove ' + units[idx].code + '?')) {

                units.splice(idx, 1);

                renderAll();

            }

        }


        // Render everything

        function renderAll() {

                document.getElementById('views-section').style.display = 'block';

                renderGapAnalysis();

                renderBloomsView();

                renderAssessmentView();

            } else {

                document.getElementById('views-section').style.display = 'none';

            }

            document.getElementById('unit-count').textContent = units.length;

        }

        // Render IPA Matrix

        function renderIPAMatrix() {

            const sorted = [...units].map((u, i) => ({ ...u, originalIdx: i }));

            sorted.sort((a, b) => a.level - b.level);


            // Group by level

            const levels = [...new Set(sorted.map(u => u.level))].sort((a, b) => a - b);


            let html = '<table class="ipa-matrix"><thead>';


            // Level group header

            html += '<tr><th></th>';

            levels.forEach(level => {

                const unitsAtLevel = sorted.filter(u => u.level === level);

                html += `<th class="level-header" colspan="${unitsAtLevel.length}">${level}-level</th>`;

            });

            html += '</tr>';


            // Unit code header

            html += '<tr><th>CLO</th>';

            sorted.forEach(u => {

                html += `<th title="${escapeHtml(u.name)}">${escapeHtml(u.code)}</th>`;

            });

            html += '</tr></thead><tbody>';


            // CLO rows

            clos.forEach((clo, cloIdx) => {

                const shortText = clo.text.length > 50 ? clo.text.substring(0, 50) + '...' : clo.text;

                html += `<tr><td title="${escapeHtml(clo.text)}"><strong>CLO ${clo.num}.</strong> ${escapeHtml(shortText)}</td>`;

                sorted.forEach(u => {

                    const ipa = u.ipaMapping[cloIdx] || '';

                    let cellHtml = '';

                    if (ipa) {

                        const cls = ipa === 'I' ? 'ipa-cell-i' : ipa === 'P' ? 'ipa-cell-p' : 'ipa-cell-a';

                        cellHtml = `<span class="ipa-cell ${cls}" onclick="cycleIPA(${u.originalIdx},${cloIdx})" title="Click to change">${ipa}</span>`;

                    } else {

                        cellHtml = `<span class="ipa-cell" onclick="cycleIPA(${u.originalIdx},${cloIdx})" title="Click to assign" style="border:1.5px dashed var(--border-color);color:var(--text-muted);font-size:0.6rem;">+</span>`;

                    }

                    html += `<td>${cellHtml}</td>`;

                });

                html += '</tr>';

            });


            html += '</tbody></table>';

            document.getElementById('ipa-matrix-content').innerHTML = html;

            document.getElementById('matrix-title').textContent = document.getElementById('course-name').value || 'IPA Matrix';

        }


        // Gap analysis

        function renderGapAnalysis() {

            const gaps = [];


            clos.forEach((clo, cloIdx) => {

                const stages = { I: false, P: false, A: false };

                units.forEach(u => {

                    const ipa = u.ipaMapping[cloIdx];

                    if (ipa) stages[ipa] = true;

                });


                if (!stages.I) gaps.push(`CLO ${clo.num} is never <strong>Introduced</strong> — students aren't given foundational exposure.`);

                if (!stages.P) gaps.push(`CLO ${clo.num} is never <strong>Practised</strong> — students don't get opportunities to develop these skills.`);

                if (!stages.A) gaps.push(`CLO ${clo.num} is never <strong>Assured</strong> — mastery is not formally demonstrated or assessed.`);


                // Check progression order

                let introLevel = 9999, practiceLevel = 0, assuredLevel = 0;

                units.forEach(u => {

                    const ipa = u.ipaMapping[cloIdx];

                    if (ipa === 'I' && u.level < introLevel) introLevel = u.level;

                    if (ipa === 'P' && u.level > practiceLevel) practiceLevel = u.level;

                    if (ipa === 'A' && u.level > assuredLevel) assuredLevel = u.level;

                });

                if (stages.I && stages.A && introLevel >= assuredLevel) {

                    gaps.push(`CLO ${clo.num} is Introduced at ${introLevel}-level but Assured at the same or lower level — progression may be inverted.`);

                }

            });


            const panel = document.getElementById('gap-panel');

            if (gaps.length === 0) {

                panel.innerHTML = `<div class="gap-panel no-gaps"><h4>✓ No IPA gaps detected</h4><p style="font-size:0.8rem;">All CLOs have complete I → P → A coverage. Well done!</p></div>`;

            } else {

                panel.innerHTML = `<div class="gap-panel"><h4>IPA Compliance Gaps (${gaps.length})</h4>${gaps.map(g => `<div class="gap-item">${g}</div>`).join('')}</div>`;

            }

        }

// Bloom's escalation view

        function renderBloomsView() {

            const container = document.getElementById('blooms-content');

            const sorted = [...units].sort((a, b) => a.level - b.level);


            let html = '';

            clos.forEach((clo, cloIdx) => {

                html += `<div class="blooms-row"><div class="blooms-clo-label">CLO ${clo.num}</div><div class="blooms-progression">`;


                sorted.forEach(u => {

                    const ipa = u.ipaMapping[cloIdx];

                    if (!ipa) return;


                    // Find relevant ULO bloom's level for this CLO in this unit

                    let bloomLevel = 0;

                    if (u.ulos.length > 0) {

                        // Try to find the most relevant ULO

                        const cloWords = new Set(clo.text.toLowerCase().split(/\W+/).filter(w => w.length > 3));

                        let bestMatch = 0, bestBlooms = 0;

                        u.ulos.forEach(ulo => {

                            const uloWords = ulo.text.toLowerCase().split(/\W+/).filter(w => w.length > 3);

                            const overlap = uloWords.filter(w => cloWords.has(w)).length;

                            if (overlap > bestMatch || (overlap === bestMatch && ulo.bloomsLevel > bestBlooms)) {

                                bestMatch = overlap;

                                bestBlooms = ulo.bloomsLevel;

                            }

                        });

                        bloomLevel = bestBlooms || (u.ulos[0] ? u.ulos[0].bloomsLevel : 0);

                    }


                    // If no ULO bloom's found, infer from IPA stage

                    if (bloomLevel === 0) {

                        if (ipa === 'I') bloomLevel = 1; // Remember/Understand

                        else if (ipa === 'P') bloomLevel = 3; // Apply/Analyse

                        else bloomLevel = 5; // Evaluate/Create

                    }


                    const levelClass = 'blooms-level-' + bloomLevel;

                    const label = getBloomsLabel(bloomLevel);

                    html += `<span class="blooms-unit-tag ${levelClass}" title="${escapeHtml(u.name)} (${ipa}): ${label}">

                        <strong>${u.code}</strong> ${label} (${ipa})

                    </span>`;

                });


                html += '</div></div>';

            });


            container.innerHTML = html || '<div class="empty-state"><p>Add units with ULOs to see Bloom\'s progression.</p></div>';

        }


        // Assessment alignment view

        function renderAssessmentView() {

            const container = document.getElementById('assessment-content');

            const sorted = [...units].sort((a, b) => a.level - b.level);


            let html = '';

            clos.forEach((clo, cloIdx) => {

                const shortText = clo.text.length > 60 ? clo.text.substring(0, 60) + '...' : clo.text;

                html += `<div class="assessment-row">`;

                html += `<div class="assessment-clo">CLO ${clo.num}. ${escapeHtml(shortText)}</div>`;

                html += `<div class="assessment-units">`;


                let hasAssured = false;

                sorted.forEach(u => {

                    const ipa = u.ipaMapping[cloIdx];

                    if (!ipa) return;


                    if (ipa === 'A') {

                        hasAssured = true;

                        // Check if unit has assessment info

                        if (u.assessments && u.assessments.length > 0) {

                            html += `<span class="assessment-unit-tag assured">${u.code}: ${escapeHtml(u.assessments[0])}</span>`;

                        } else {

                            html += `<span class="assessment-unit-tag assured">${u.code} (Assured) ✓</span>`;

                        }

                    } else {

                        html += `<span class="assessment-unit-tag">${u.code} (${ipa})</span>`;

                    }

                });


                if (!hasAssured) {

                    html += `<span class="assessment-unit-tag no-assessment">⚠ No unit assures this CLO</span>`;

                }


                html += '</div></div>';

            });


            container.innerHTML = html || '<div class="empty-state"><p>Load CLOs and add units to see assessment alignment.</p></div>';

        }
